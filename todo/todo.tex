\usepackage{xcolor}

\newcommand{\todo}[1]{}
\newcommand{\fixme}[1]{}

\newcommand{\enabletodo}{
    \renewcommand{\todo}[1]{ {
        {   
            \color{blue}
            \fontfamily{\sfdefault}
            \fontseries{b}
            \selectfont
            TODO
        } ##1
    } }

    \renewcommand{\fixme}[1]{ {
        {   
            \color{red}
            \fontfamily{\sfdefault}
            \fontseries{b}
            \selectfont
            FIXME
        } ##1
    } }
}

\newcommand{\disabletodo}[0]{
    \renewcommand{\todo}[1]{}
    \renewcommand{\fixme}[1]{}
}

\enabletodo
