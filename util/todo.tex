\usepackage{xcolor}
\definecolor{darkgreen}{HTML}{008800}

\newcommand{\todo}[1]{}
\newcommand{\fixme}[1]{}
\newcommand{\pic}[1]{}

\newcommand{\enabletodo}{
    \renewcommand{\todo}[1]{ {
        {   
            \color{blue}
            \fontfamily{\sfdefault}
            \fontseries{b}
            \selectfont
            TODO
        } ##1
    } }

    \renewcommand{\fixme}[1]{ {
        {   
            \color{red}
            \fontfamily{\sfdefault}
            \fontseries{b}
            \selectfont
            FIXME
        } ##1
    } }

    \renewcommand{\pic}[1]{ {
        {   
            \color{darkgreen}
            \fontfamily{\sfdefault}
            \fontseries{b}
            \selectfont
            PICTURE
        } ##1
    } }
}

\newcommand{\disabletodo}[0]{
    \renewcommand{\todo}[1]{}
    \renewcommand{\fixme}[1]{}
    \renewcommand{\pic}[1]{}
}

\enabletodo
